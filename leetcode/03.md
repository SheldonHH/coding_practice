#include<vector>
#include<string>
#include<algorithm>

class Solution {
public:
    int lengthOfLongestSubstring(std::string s) {
        if (s.empty()) {
            return 0;
        }
      
		//int map[256]; std::fill(map, map+256, -1);
        std::vector<int> map(256, -1);
        int len = 0;
        int pre = -1;
        int cur = 0;
        for (int i = 0; i < s.size(); i++) {
            pre = std::max(pre, map[s[i]]);
            cur = i - pre;
            len = std::max(len, cur);
            map[s[i]] = i;
        }
        return len;
    }
};
